209function importFoodsCsv(text, target = foodsData) {
210 const summary = {
211 addedCities: new Set(),
212 touchedCities: new Set(),
213 addedItems: 0,
214 updatedItems: 0,
215 errors: []
216 };
217
218 if (!text) return summary;
219
220 const lines = String(text).split(/\r?\n/);
221 lines.forEach((line, index) => {
222 const normalizedLine = line.replace(/\uFEFF/g, '').trim();
223 if (!normalizedLine) return;
224 if (index === 0 && /city/i.test(normalizedLine) && /food/i.test(normalizedLine) && /price/i.test(normalizedLine)) {
225 return;
226 }
227 const parts = normalizedLine.split(',');
228 if (parts.length < 3) {
229 summary.errors.push(`Line ${index + 1}: expected "City,Food Item,Price".`);
230 return;
231 }
232 const city = parts.shift().trim();
233 const priceRaw = parts.pop().trim();
234 const itemName = parts.join(',').trim();
235 if (!city || !itemName || !priceRaw) {
236 summary.errors.push(`Line ${index + 1}: missing city, food, or price.`);
237 return;
238 }
239 const priceNumber = parsePriceValue(priceRaw);
240 if (priceNumber === null) {
241 summary.errors.push(`Line ${index + 1}: invalid price "${priceRaw}".`);
242 return;
243 }
244 if (!target[city]) {
245 target[city] = [];
246 summary.addedCities.add(city);
247 }
248 const cityItems = target[city];
249 const existing = cityItems.find(entry => entry.item.toLowerCase() === itemName.toLowerCase());
250 const formattedPrice = formatUSD(priceNumber);
251 if (existing) {
252 existing.price = formattedPrice;
253 summary.updatedItems++;
254 } else {
255 cityItems.push({ item: itemName, price: formattedPrice });
256 cityItems.sort((a, b) => a.item.localeCompare(b.item));
257 summary.addedItems++;
258 }
259 summary.touchedCities.add(city);
260 });
261
262 return summary;
263}

1004// 2-opt improvement for an open path (keeps first node fixed)
1005function twoOptOpen(order, M, maxIter=2000) {
1006 const n = order.length;
1007 function pathLen(ord) {
1008 let s=0; for (let i=0;i<n-1;i++) s += M[ord[i]][ord[i+1]]; return s;
1009 }
1010 let best = order.slice();
1011 let bestLen = pathLen(best);
1012 let improved = true, iter=0;
1013 while (improved && iter < maxIter) {
1014 improved = false; iter++;
1015 for (let i=1;i<n-2;i++) {
1016 for (let k=i+1;k<n-1;k++) {
1017 const a=best[i-1], b=best[i], c=best[k], d=best[k+1];
1018 const delta = (M[a][c] + M[b][d]) - (M[a][b] + M[c][d]);
1019 if (delta < -1e-6) {
1020 const seg = best.slice(i, k+1).reverse();
1021 best.splice(i, k-i+1, ...seg);
1022 bestLen += delta;
1023 improved = true;
1024 }
1025 }
1026 }
1027 }
1028 return best;
1029}

1361// 2-opt improvement for open path (keeps first node fixed)
1362function LON_twoOpt(order, M, maxIter=1500) {
1363 const n = order.length;
1364 function pathLen(ord) { let s=0; for (let i=0;i<n-1;i++) s += M[ord[i]][ord[i+1]]; return s; }
1365 let best = order.slice();
1366 let bestLen = pathLen(best);
1367 let improved = true, iter = 0;
1368 while (improved && iter < maxIter) {
1369 improved = false; iter++;
1370 for (let i=1;i<n-2;i++) {
1371 for (let k=i+1;k<n-1;k++) {
1372 const a=best[i-1], b=best[i], c=best[k], d=best[k+1];
1373 const delta = (M[a][c] + M[b][d]) - (M[a][b] + M[c][d]);
1374 if (delta < -1e-6) {
1375 const seg = best.slice(i, k+1).reverse();
1376 best.splice(i, k-i+1, ...seg);
1377 bestLen += delta;
1378 improved = true;
1379 }
1380 }
1381 }
1382 }
1383 return best;
1384}
